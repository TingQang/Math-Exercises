% 第2章：一元函数微分学
\chapter{一元函数微分学}
\label{ch:differential-calculus}

本章介绍导数与微分的概念、性质和应用，包括导数的定义、计算方法、几何意义，以及函数的单调性、极值、最值等性质。

% 题型2.1 导数的定义与计算
\begin{question-type}{导数的定义与计算}
本题型主要考查导数的定义、基本导数公式、复合函数求导法则、隐函数求导、参数方程求导以及高阶导数的计算。

\begin{exercise}
计算以下函数的导数：
\begin{enumerate}[label={【\arabic*】}]
    \item \difficulty{1} $y = x^3 + 2x^2 - 3x + 1$ \exercisesource{李永乐复习全书P145例2}
    \item \difficulty{1} $y = \sin x \cos x$ \exercisesource{2024考研数学一真题}
    \item \difficulty{2} $y = e^x \ln x$ \exercisesource{B站 BV1xx4y1E7xx}
    \item \difficulty{2} $y = \frac{x^2 + 1}{x - 1}$ \exercisesource{教材习题3.1}
    \item \difficulty{3} $y = \arctan \frac{1}{x}$ \exercisesource{知网 https://kns.cnki.net/...}
\end{enumerate}
\end{exercise}

\begin{exercise}
求以下隐函数的导数：
\begin{enumerate}[label={【\arabic*】}]
    \item \difficulty{2} $x^2 + y^2 = 1$ \exercisesource{李永乐复习全书P152}
    \item \difficulty{3} $e^y + xy = 0$ \exercisesource{2023考研数学一真题}
    \item \difficulty{4} $\sin(x + y) = x^2 y$ \exercisesource{综合题型训练}
\end{enumerate}
\end{exercise}

\begin{exercise}
求参数方程所确定的函数的导数：
\begin{enumerate}[label={【\arabic*】}]
    \item \difficulty{3} $\begin{cases} x = t^2 + 1 \\ y = t^3 - t \end{cases}$ \exercisesource{参数方程专项}
    \item \difficulty{4} $\begin{cases} x = e^t \cos t \\ y = e^t \sin t \end{cases}$ \exercisesource{高等数学教材P178}
\end{enumerate}
\end{exercise}
\end{question-type}

% 题型2.2 函数的单调性、极值与最值
\begin{question-type}{函数的单调性、极值与最值}
本题型主要考查利用导数判断函数的单调性、求极值点和极值、求函数在区间上的最值，以及闭区间上连续函数的最值定理应用。

\begin{exercise}
判断函数的单调性：
\begin{enumerate}[label={【\arabic*】}]
    \item \difficulty{2} $f(x) = x^3 - 3x$ \exercisesource{单调性基础题}
    \item \difficulty{3} $f(x) = e^x - x$ \exercisesource{2024考研数学一真题}
    \item \difficulty{2} $f(x) = \frac{x}{1 + x^2}$ \exercisesource{李永乐复习全书P165}
\end{enumerate}
\end{exercise}

\begin{exercise}
求函数的极值：
\begin{enumerate}[label={【\arabic*】}]
    \item \difficulty{2} $f(x) = x^3 - 3x^2 + 1$ \exercisesource{极值计算题}
    \item \difficulty{3} $f(x) = \frac{x^2}{x^2 + 1}$ \exercisesource{B站 BV1xx4y1E7xx}
    \item \difficulty{3} $f(x) = e^{-x^2}$ \exercisesource{高等数学教材}
\end{enumerate}
\end{exercise}

\begin{exercise}
求函数在给定区间上的最值：
\begin{enumerate}[label={【\arabic*】}]
    \item \difficulty{1} $f(x) = x^2 - 2x + 1$ 在 $[0, 3]$ 上 \exercisesource{最值基础题}
    \item \difficulty{3} $f(x) = \sin x + \cos x$ 在 $[0, 2\pi]$ 上 \exercisesource{三角函数最值}
    \item \difficulty{3} $f(x) = x e^{-x}$ 在 $(-\infty, +\infty)$ 上 \exercisesource{2023考研数学一真题}
\end{enumerate}
\end{exercise}
\end{question-type}

% 题型2.3 不等式证明
\begin{question-type}{不等式证明}
本题型主要考查利用导数证明不等式，包括利用单调性证明不等式、利用极值证明不等式等。

\begin{exercise}
利用导数证明不等式：
\begin{enumerate}[label={【\arabic*】}]
    \item \difficulty{2} 当 $x > 0$ 时，$e^x > 1 + x$ \exercisesource{不等式证明经典题}
    \item \difficulty{2} 当 $x > 0$ 时，$\ln(1 + x) < x$ \exercisesource{李永乐复习全书P172}
    \item \difficulty{4} 当 $0 < x < 1$ 时，$\sin x > x - \frac{x^3}{6}$ \exercisesource{高等数学教材}
\end{enumerate}
\end{exercise}

\begin{exercise}
利用函数的单调性证明不等式：
\begin{enumerate}[label={【\arabic*】}]
    \item \difficulty{2} 当 $x > 1$ 时，$x + \frac{1}{x} \geq 2$ \exercisesource{均值不等式}
    \item \difficulty{4} 当 $x > 0$ 时，$\frac{x}{1 + x} < \ln(1 + x)$ \exercisesource{积分不等式}
\end{enumerate}
\end{exercise}
\end{question-type}
